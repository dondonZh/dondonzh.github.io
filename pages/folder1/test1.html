    # 函数防抖和节流

    # 1.函数防抖

    连续小于规定时间执行最后只触发一次 常用于（防止多次点击，重复提交）

    # 实现

    /**********************防抖**********************************************************/
/**
 * @description: 防抖函数：函数被触发 n 秒后再执行回调，如果在这 n 秒内又被触发，则重新计时
 * @param {Function} fn 要执行的函数
 * @param {Number} wait  wait毫秒后执行回调
 * @param {Boolean} immediate 表示是否立即执行
 * @param {*} ...arr 传递给fn的参数
 */
function debounce(fn, wait, immediate, ...arr) {
  let timer = null;
  return () => {
    if (timer) {
      // 如果有一个函数在等待执行 清除定时器 下面重新计时
      clearTimeout(timer);
      timer = null; // 清空timer 下次重启定时器
    }

    if (immediate) {
      let callNow = !timer;
      timer = setTimeout(() => {
        timer = null;
      }, wait);
      if (callNow) fn(...arr);
    } else {
      // 设定时器/重置定时器
      timer = setTimeout(() => {
        fn(...arr); // wait时间后 执行回调 期间再触发debounce 需要重新等待
      }, wait);
    }
  };
}
// 要防抖的函数
let actionFn = function(a, b) {
  console.log("防抖回调", a, b);
};
const afterDebounceFunctuion = debounce(
  actionFn,
  1500,
  true,
  "actionFn参数1",
  "参数2"
);
setInterval(afterDebounceFunctuion, 1000);
//这里可以点击调用或者 setInternal调用

    # 1.函数节流

    连续小于规定时间执行最后按规定时间连续触发 常用于（刷新率）

    # 实现

    /**********************节流************************************************/
/**
 * @description: 节流函数：规定一个单位时间，在这个单位时间内，只能有一次触发事件的回调函数执行
 * @param {Function} fn 要执行的函数
 * @param {Number} gapTime  单位时间
 * @param {Number} type   type 1 表时间戳版，2 表定时器版
 * @param {*} ...arr 传递给fn的参数
 */
function throttle(fn, gapTime, type, ...arr) {
  if (type === 1) {
    var previous = 0;
  } else if (type === 2) {
    var timeout;
  }
  return () => {
    if (type === 1) {
      let nowTime = Date.now(); // 当前时间
      // 当前时间-上次执行的时间是否超过间隔时间 就执行回调
      if (nowTime - previous > gapTime) {
        fn(...arr); // ...arr为fn的参数
        previous = nowTime; // 重置上次执行时间为当前时间 方便下次执行
      }
    } else if (type === 2) {
      if (!timeout) {
        timeout = setTimeout(() => {
          timeout = null;
          fn(...arr); // ...arr为fn的参数
        }, gapTime);
      }
    }
  };
}
let actionThFn = (a, b) => {
  console.log("节流回调", a, b); // 要执行的函数
};
const afterThrottleFunctuion = throttle(
  actionThFn,
  1000,
  2,
  "actionFn参数1",
  "参数2"
);
// 这里的afterThrottleFunctuion同理可用来绑定点击或者其他事件
// setInterval(afterThrottleFunctuion, 10);
// 每隔10毫秒都会触发一次throttle，每隔一秒触发一次actionFn回调(1秒内再次触发被丢弃
